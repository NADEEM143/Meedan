function updateCount() { const names = getNames(); document.getElementById('counter').innerText = `--- Active: ${names.length} ---`; localStorage.setItem('meedan_names', document.getElementById('nameList').value); } function getNames() { return document.getElementById('nameList').value.split('\n').map(n => n.trim()).filter(n => n !== ""); } function generateVerifyCode() { return 'MD-' + Math.random().toString(36).substr(2, 6).toUpperCase(); } function startRoll() { const names = getNames(); if (names.length === 0) return alert("Please enter names!"); if (audioCtx.state === 'suspended') audioCtx.resume(); const spinBtn = document.getElementById('rollBtn'); spinBtn.disabled = true; clearInterval(countdownInterval); document.getElementById('claimTimer').style.display = "none"; document.getElementById('noShowBtn').style.display = "none"; document.getElementById('verifyArea').style.display = "none"; document.getElementById('congrats').style.display = "none"; document.getElementById('status').innerText = "CALCULATING..."; let delay = 50, time = 0; const roll = () => { const r = Math.floor(Math.random() * names.length); document.getElementById('result').innerText = names[r]; playTick(); time += delay; if (time < 6000) delay = 50; else if (time < 8500) delay += 20; else delay += 60; if (time < 10000) setTimeout(roll, delay); else { finalizeWinner(names[r]); spinBtn.disabled = false; } }; roll(); } function finalizeWinner(winner) { winCount++; const vCode = generateVerifyCode(), timeStr = new Date().toLocaleTimeString(); let prize = winCount === 1 ? localStorage.getItem('m_p1') : winCount === 2 ? localStorage.getItem('m_p2') : localStorage.getItem('m_p3'); document.getElementById('congrats').style.display = "block"; document.getElementById('result').innerText = "🏆 " + winner; document.getElementById('result').className = "winner-text"; document.getElementById('verifyCode').innerText = vCode; document.getElementById('verifyArea').style.display = "block"; document.getElementById('removeWinnerBtn').style.display = "block"; document.getElementById('status').innerText = "WINNER VERIFIED"; startClaimTimer(); winnersArray.push({ "Rank": winCount, "Name": winner, "Prize": prize || "Winner", "Code": vCode, "Time": timeStr }); localStorage.setItem('meedan_winners', JSON.stringify(winnersArray)); refreshHistoryUI(); } function startClaimTimer() { let seconds = 180; const timerDisp = document.getElementById('claimTimer'), noShowBtn = document.getElementById('noShowBtn'); timerDisp.style.display = "block"; noShowBtn.style.display = "block"; countdownInterval = setInterval(() => { let m = Math.floor(seconds / 60), s = seconds % 60; timerDisp.innerText = `CLAIM TIME: ${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`; if (seconds <= 0) { clearInterval(countdownInterval); timerDisp.innerText = "EXPIRED!"; playAlert(); } seconds--; }, 1000); } function noShowRespin() { if(confirm("Re-spin?")) { removeWinner(); startRoll(); } } function refreshHistoryUI() { const hl = document.getElementById('historyList'); hl.innerHTML = ""; [...winnersArray].reverse().forEach(win => { let color = win.Rank === 1 ? "var(--gold)" : win.Rank === 2 ? "var(--silver)" : win.Rank === 3 ? "var(--bronze)" : "white"; const li = document.createElement('li'); li.innerHTML = `
🏆 #${win.Rank} ${win.Name} (${win.Prize})
ID: ${win.Code} | ${win.Time}
`; hl.appendChild(li); }); } function copyCode() { navigator.clipboard.writeText(document.getElementById('verifyCode').innerText); alert("Copied!"); } function removeWinner() { const wt = document.getElementById('result').innerText.replace("🏆 ", ""); document.getElementById('nameList').value = getNames().filter(n => n !== wt).join('\n'); document.getElementById('removeWinnerBtn').style.display = "none"; document.getElementById('congrats').style.display = "none"; document.getElementById('result').innerText = "---"; document.getElementById('claimTimer').style.display = "none"; document.getElementById('noShowBtn').style.display = "none"; updateCount(); } function clearHistory() { if (confirm("Reset?")) { winnersArray = []; winCount = 0; localStorage.removeItem('meedan_winners'); refreshHistoryUI(); } } function downloadTxt(f, t) { const e = document.createElement('a'); e.setAttribute('href', 'data:text/plain;charset=utf-8,' + encodeURIComponent(t)); e.setAttribute('download', f); e.style.display = 'none'; document.body.appendChild(e); e.click(); document.body.removeChild(e); } function exportParticipantsExcel() { downloadTxt("Contestants.txt", getNames().join('\n')); } function exportWinnersExcel() { downloadTxt("Log.txt", winnersArray.map(w => `#${w.Rank} - ${w.Name} (${w.Prize}) | ID: ${w.Code}`).join('\n')); }
